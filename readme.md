

## Usage
> For more use-cases see the [tests](./test.js)

```js
var gulp = require('gulp');
var micromatch = require('gulp-micromatch');

var opts = {
  dot: true,
  matchBase: true
};

gulp.src(['*.md', '*.js', '*.txt', '*.json'])
  .pipe(micromatch('*.(js|json)', opts))
  .pipe(through2.obj(function(file, enc, next) {
    // a.js
    // b.json
    // c.json
  }));
```
